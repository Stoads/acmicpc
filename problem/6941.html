솔직히 그냥 달력 출력하면 끝나는 문제라 설명할게 없다.


#include <cstdio>
int n,d;
int main(){
    scanf("%d%d",&n,&d);
    for(int i=0;i<7;i++)
        printf("%.3s ","SunMonTueWedThrFriSat"+i*3);
    for(int i=0;i<6*printf("\n")&&1-n<=d;i++)
        for(int j=0;j<7;j++)
            --n>0?printf("    ") : 1-n>d||printf("%3d ",1-n);
}

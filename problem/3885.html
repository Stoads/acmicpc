Solved.ac 3번째 마라톤 참여 7번째 문제

문제 자체는 굉장히 쉬운 구현 문제이다. 근데 소숫점 쓰는거며 귀찮은게 많다.. 무엇보다 소수로 출력이 떠야하는 문제는 오차가 발생할수 있는점 때문에 쉽지 않다..


#include <cstdio>
#include <algorithm>
int sol(){
    int n,m;
    int bucket[12]={};
    scanf("%d%d",&n,&m);
    if(n==0)return 0;
    int _max=0,sz=0;
    for(int i=0,x;i<n;i++){
        scanf("%d",&x);
        _max=std::max(_max,++bucket[x/m]);
        sz=std::max(sz,x/m);
    }
    double res=0.01;
    for(int i=0;i<sz;i++)
        res+=(double)bucket[i]/_max*(sz-i)/sz;
    printf("%.10g\n",res);
    return 1;
}
int main(){
    while(sol());
}

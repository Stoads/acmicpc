간단한 DP로 확률 계산하는 문제. 1이하의 계산에서 소숫점 3째자리 까지니까 아마 그냥 계산해도 부동소숫점 문제는 발생하지 않는다.(웬만해서)


#include <cstdio>
int n;
int x,y;
double dp[123][123];
double dice(int a,int b){
    if(dp[a][b]>=0)return dp[a][b];
    if(b==0)return dp[a][b]=1;
    if(a==0)return dp[a][b]=0;
    return dp[a][b]=dice(a,b-1)/10+dice(a-1,b-1)*2/10+dice(a-1,b)*7/10;
}
double solve(){
    scanf("%d%d",&x,&y);
    for(int i=0;i<123;i++)
        for(int j=0;j<123;j++)
            dp[i][j]=-1;
    return dice(x,y);
}
int main(){
    for(scanf("%d",&n);n--;printf("%.3lf\n",solve()));
}

솔직히 말하면 double로 진행되는 문제는 아무래도 부동소숫점때문에 별로 좋아하진 않는다. 머 그래도 별 수 있나

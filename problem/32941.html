간단하게 요약해서 N개의 배열에 대해서 각 배열이 전부 X를 포함하는가? 이다.


#include <cstdio>
int T,X;
int N;
int K;
int main(){
    scanf("%d%d",&T,&X);
    scanf("%d",&N);
    int res=1;
    for(int i=0;i<N;i++){
        scanf("%d",&K);
        int flg=0;
        for(int j=0,p;j<K;j++){
            scanf("%d",&p);
            flg=flg||p==X;
        }
        res&=flg;
    }
    printf(res?"YES":"NO");
}
